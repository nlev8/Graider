# Phase 4.1 — Portal Grading → Celery Migration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Migrate join-code portal grading from daemon threads to Celery + Redis task queue on a separate Railway worker dyno, so student submissions survive Railway deploys, worker crashes, and concurrent load.

**Architecture:** Celery app reads `CELERY_BROKER_URL` (defaults to `${REDIS_URL}/1`). New Railway worker service runs `celery worker` with prefork pool, concurrency 8. Task re-fetches submission state from Supabase and runs existing grading pipeline via an extracted pure function `grade_portal_submission_sync`. Row-level dedup via `grading_task_id` + `grading_started_at` columns prevents double-grading across retries and crash redeliveries. Feature flag `CELERY_PORTAL_GRADING` allows zero-code rollback.

**Tech Stack:** Python 3.14, Celery 5.x, Redis (existing Railway), Flask, Supabase (supabase-py 2.27.2), Sentry SDK with CeleryIntegration, pytest.

**Spec:** `docs/superpowers/specs/2026-04-16-phase4.1-portal-celery-design.md`

- [ ] **Step 2.8: Create `backend/celery_app.py`**

```python
"""Phase 4.1 Celery application for Graider.

Instantiates the Celery app, reads CELERY_BROKER_URL from env (fails fast if unset),
and wires the Sentry CeleryIntegration at worker process init.

Task discovery: autodiscover_tasks(['backend.tasks']).
"""
import logging
import os

from celery import Celery
from celery.signals import worker_process_init

_logger = logging.getLogger(__name__)

_broker_url = os.environ.get('CELERY_BROKER_URL')
if not _broker_url:
    raise RuntimeError(
        "CELERY_BROKER_URL env var is required to start the Celery worker. "
        "Set it on the Railway worker service (default: redis://...:6379/1 to keep "
        "flask-limiter/flask-session on DB 0 undisturbed)."
    )

celery_app = Celery('graider')
celery_app.conf.broker_url = _broker_url
celery_app.conf.task_serializer = 'json'
celery_app.conf.accept_content = ['json']
celery_app.conf.result_backend = None  # Supabase row is source of truth
celery_app.conf.worker_send_task_events = True
celery_app.conf.task_send_sent_event = True

# Auto-discover tasks in backend.tasks package
celery_app.autodiscover_tasks(['backend.tasks'])


@worker_process_init.connect
def _init_worker_process(**kwargs):
    """Initialize Sentry + rebuild Supabase client on each worker process spawn.

    Prefork pool forks fresh processes; each needs its own Sentry init.
    supabase-py uses httpx.Client whose connection pool is not fork-safe —
    importing the client module in the worker process rebuilds it cleanly.
    """
    from backend.observability.sentry import init_sentry
    init_sentry(environment='worker')
    _logger.info("Celery worker process init complete (Sentry + Supabase client rebuilt)")
```

```python
def get_api_key(
    provider: str,
    user_id: str,
    *,
    district_id: str | None = None,
) -> str | None:
    """Resolve API key for provider.

    Resolution order:
    1. Per-teacher BYOK (teacher_id=user_id)
    2. District BYOK (district_id passed explicitly OR pulled from flask.g)
    3. Env fallback (OPENAI_API_KEY etc.)

    Celery workers pass district_id explicitly (no Flask context).
    Flask routes can omit it → falls back to flask.g.district_id (backward compat).
    """
    # Existing teacher lookup logic stays unchanged
    teacher_key = _fetch_teacher_key(provider, user_id)
    if teacher_key:
        return teacher_key

    # District lookup — use explicit district_id OR fall back to flask.g
    if district_id is None:
        try:
            from flask import g as _flask_g
            district_id = getattr(_flask_g, 'district_id', None)
        except (RuntimeError, ImportError):
            district_id = None  # No Flask context (Celery worker)

    if district_id:
        district_key = _fetch_district_key(provider, district_id)
        if district_key:
            return district_key

    # Env fallback
    return os.environ.get(f'{provider.upper()}_API_KEY')

        # ... whatever existing providers the function returns
```

```python
"""Phase 4.1 portal grading Celery task."""
import logging

import sentry_sdk
from celery import Task

from backend.celery_app import celery_app

_logger = logging.getLogger(__name__)


class TransientError(Exception):
    """Transient failures that Celery should retry (OpenAI 5xx, Supabase 503)."""


class PortalGradingTask(Task):
    """Base task class with on_failure hook for terminal 'failed' state."""

    def on_failure(self, exc, task_id, args, kwargs, einfo):
        """Called by Celery after max_retries is exhausted. Mark submission row terminal."""
        submission_id = args[0] if args else kwargs.get('submission_id')
        supabase_table = args[2] if len(args) > 2 else kwargs.get('supabase_table', 'submissions')
        if not submission_id:
            return
        try:
            from backend.services.portal_grading import update_submission
            update_submission(
                supabase_table=supabase_table,
                submission_id=submission_id,
                status='failed',
                error_message=str(exc)[:500],
            )
        except Exception:
            # Sentry already has the original exception; don't mask it.
            pass


@celery_app.task(
    base=PortalGradingTask,
    name='grading.portal_submission',
    bind=True,
    acks_late=True,
    autoretry_for=(TransientError, TimeoutError, ConnectionError),
    retry_backoff=True,
    retry_backoff_max=300,
    max_retries=3,
    time_limit=900,
    soft_time_limit=840,
)
def grade_portal_submission(
    self,
    submission_id: str,
    teacher_id: str,
    supabase_table: str,
    *,
    district_id: str | None = None,
    user_id: str | None = None,
) -> None:
    """Phase 4.1 join-code portal grading task.

    Exactly-once semantics via row-level task_id + grading_started_at dedup.
    Fetches submission state from Supabase and runs the pure grading pipeline.
    """
    if user_id:
        sentry_sdk.set_user({"id": user_id})
    if district_id:
        sentry_sdk.set_tag("district", district_id)

    from backend.services.portal_grading import (
        grade_portal_submission_sync,
        fetch_submission_full_context,
    )

    ctx = fetch_submission_full_context(supabase_table, submission_id, teacher_id)
    if not ctx:
        _logger.warning("Submission not found for grading: %s", submission_id)
        return

    grade_portal_submission_sync(
        submission_id=submission_id,

        student_info=ctx['student_info'],
        teacher_config=ctx['teacher_config'],
        teacher_id=teacher_id,
```
